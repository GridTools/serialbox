32 static std::string vecToString(VecType&& vec) {
33  std::stringstream ss;
34  if(!vec.empty()) {
35  for(std::size_t i = 0; i < vec.size() - 1; ++i)
36  ss << vec[i] << ", ";

48 serialboxSerializer_t* serialboxSerializerCreate(int mode, const char* directory,
49  const char* prefix, const char* archive) {
50  serialboxSerializer_t* serializer = allocate<serialboxSerializer_t>();
51 
52  try {
53  switch(mode) {
54  case Read:
55  serializer->impl = new Serializer(serialbox::OpenModeKind::Read, directory, prefix, archive);
56  break;
57  case Write:
58  serializer->impl = new Serializer(serialbox::OpenModeKind::Write, directory, prefix, archive);
59  break;
60  case Append:
61  serializer->impl =
62  new Serializer(serialbox::OpenModeKind::Append, directory, prefix, archive);
63  break;
64  default:
65  serialbox_unreachable((boost::format("invalid mode (%i)") % mode).str().c_str());
66  }
67  serializer->ownsData = 1;
68  } catch(std::exception& e) {
69  std::free(serializer);
70  serializer = NULL;
71  serialboxFatalError(e.what());
72  }
73  return serializer;
74 }

163  const Serializer* ser = toConstSerializer(serializer);
164  const auto& savepointVector = ser->savepointVector().savepoints();
165 
166  serialboxSavepoint_t** savepoints = allocate<serialboxSavepoint_t*>(savepointVector.size());
167 
168  if(!savepoints)

170 
171  for(std::size_t i = 0; i < savepointVector.size(); ++i) {
172  serialboxSavepoint_t* savepoint = allocate<serialboxSavepoint_t>();
173  savepoint->impl = savepointVector[i].get();
174  savepoint->ownsData = 0;
175  savepoints[i] = savepoint;
176  }
177 
178  return savepoints;

231 int serialboxSerializerAddField2(serialboxSerializer_t* serializer, const char* name, int type,
232  int bytesPerElement, int iSize, int jSize, int kSize, int lSize,
233  int iMinusHalo, int iPlusHalo, int jMinusHalo, int jPlusHalo,
234  int kMinusHalo, int kPlusHalo, int lMinusHalo, int lPlusHalo) {
235  Serializer* ser = toSerializer(serializer);
236 
237  try {
238  serialbox::TypeID typeID = static_cast<serialbox::TypeID>((int)type);
239  std::string typeName = serialbox::TypeUtil::toString(typeID);
240 
241  if(bytesPerElement != serialbox::TypeUtil::sizeOf(typeID))
242  throw serialbox::Exception("inconsistent bytes-per-element: got '%i' but according to passed "
243  "type '%s' expected '%i'",
244  bytesPerElement, typeName, serialbox::TypeUtil::sizeOf(typeID));
245 
246  int rank =
247  (iSize != 1 ? 1 : 0) + (jSize != 1 ? 1 : 0) + (kSize != 1 ? 1 : 0) + (lSize != 1 ? 1 : 0);
248 
249  std::vector<int> dims{iSize, jSize, kSize, lSize};
250  MetainfoMap metaInfo;
251  metaInfo.insert("__name", name);
252  metaInfo.insert("__elementtype", typeName);
253  metaInfo.insert("__bytesperelement", bytesPerElement);
254  metaInfo.insert("__rank", rank);
255  metaInfo.insert("__isize", iSize);
256  metaInfo.insert("__jsize", jSize);
257  metaInfo.insert("__ksize", kSize);
258  metaInfo.insert("__lsize", lSize);
259  metaInfo.insert("__iminushalosize", iMinusHalo);
260  metaInfo.insert("__iplushalosize", iPlusHalo);
261  metaInfo.insert("__jminushalosize", jMinusHalo);
262  metaInfo.insert("__jplushalosize", jPlusHalo);
263  metaInfo.insert("__kminushalosize", kMinusHalo);
264  metaInfo.insert("__kplushalosize", kPlusHalo);
265  metaInfo.insert("__lminushalosize", lMinusHalo);
266  metaInfo.insert("__lplushalosize", lPlusHalo);
267 
268  FieldMetainfo fieldMetainfo(typeID, dims, metaInfo);
269 
270  // Field was already registered with the same meta-data
271  if(ser->hasField(name) && (ser->getFieldMetainfoImplOf(name) == fieldMetainfo))
272  return 0;
273 
274  ser->registerField(name, fieldMetainfo);
275  } catch(std::exception& e) {
276  serialboxFatalError(e.what());
277  }
278  return 1;
279 }

321 serialbox::StorageView makeStorageView(Serializer* ser, const char* name, void* originPtr,
322  const int* strides, int numStrides) {
323 
324  // Check if field exists
325  auto it = ser->fieldMap().findField(name);
326  if(it == ser->fieldMap().end())
327  throw serialbox::Exception("field '%s' is not registerd within the Serializer", name);
328 
329  // Get necessary meta-information to construct StorageView
330  const auto& dims = it->second->dims();
331  std::vector<int> stridesVec(strides, strides + numStrides);
332 
333  if(dims.size() != stridesVec.size())
334  throw serialbox::Exception("inconsistent number of dimensions and strides of field '%s'"
335  "\nDimensions as: [ %s ]"
336  "\nStrides as: [ %s ]",
337  name, internal::vecToString(dims),
338  internal::vecToString(stridesVec));
339 
340  return serialbox::StorageView(originPtr, it->second->type(), dims, stridesVec);
341 }

375 void serialboxSerializerReadSliced(serialboxSerializer_t* serializer, const char* name,
376  const serialboxSavepoint_t* savepoint, void* originPtr,
377  const int* strides, int numStrides, const int* slice) {
378  Serializer* ser = toSerializer(serializer);
379  const Savepoint* sp = toConstSavepoint(savepoint);
380 
381  try {
382  serialbox::StorageView storageView(
383  internal::makeStorageView(ser, name, originPtr, strides, numStrides));
384 
385  serialbox::Slice sliceObj((serialbox::Slice::Empty()));
386  for(int i = 0; i < numStrides; ++i)
387  sliceObj.sliceTriples().push_back({slice[3 * i], slice[3 * i + 1], slice[3 * i + 2]});
388  storageView.setSlice(sliceObj);
389 
390  ser->read(name, *sp, storageView);
391  } catch(std::exception& e) {
392  serialboxFatalError(e.what());
393  }

333 template <typename TDataField>
334 void Serializer::WriteField(std::string name, const TDataField& field, const Savepoint& savepoint,
335  const int iStride, const int jStride, const int kStride) {
336 
337  if(name.empty())
338  name = field.name();
339 
340  // Retrieve size, boundaries and strides
341  IJKSize size;
342  size.Init(field.storage().allocatedSize().iSize(), field.storage().allocatedSize().jSize(),
343  field.storage().allocatedSize().kSize());
344 
345  IJKBoundary boundary;
346  boundary.Init(field.boundary().iMinusOffset(), field.boundary().iPlusOffset(),
347  field.boundary().jMinusOffset(), field.boundary().jPlusOffset(),
348  field.boundary().kMinusOffset(), field.boundary().kPlusOffset());
349 
350  const int bytesPerElement = sizeof(typename TDataField::ValueType);
351 
352  // Register field
353  this->RegisterField(name, type_name<typename TDataField::ValueType>(), bytesPerElement,
354  size.iSize(), size.jSize(), size.kSize(), 1, -boundary.iMinusOffset(),
355  boundary.iPlusOffset(), -boundary.jMinusOffset(), boundary.jPlusOffset(),
356  -boundary.kMinusOffset(), boundary.kPlusOffset(), 0, 0);
357 
358  this->WriteField(name, savepoint, field.storage().pStorageBase(), iStride, jStride, kStride, 0);
359 }

361 template <typename TDataField>
362 void Serializer::ReadField(std::string name, TDataField& field, const Savepoint& savepoint,
363  bool hasStorageInI, bool hasStorageInJ, const int iStride,
364  const int jStride, const int kStride, bool alsoPrevious) const {
365  typedef typename TDataField::ValueType ValueType;
366 
367  if(name.empty())
368  name = field.name();
369 
370  // Get info of serialized field
371  const DataFieldInfo& info = FindField(name);
372 
373  // Check size and boundaries
374  IJKSize size;
375  size.Init(field.storage().allocatedSize().iSize(), field.storage().allocatedSize().jSize(),
376  field.storage().allocatedSize().kSize());
377 
378  IJKBoundary boundary;
379  boundary.Init(field.boundary().iMinusOffset(), field.boundary().iPlusOffset(),
380  field.boundary().jMinusOffset(), field.boundary().jPlusOffset(),
381  field.boundary().kMinusOffset(), field.boundary().kPlusOffset());
382 
383  if((info.iSize() != size.iSize()) || (info.jSize() != size.jSize()) ||
384  (info.kSize() != size.kSize()) || (info.lSize() != 1)) {
385  // Throw exception
386  std::ostringstream errorstr;
387  errorstr << "Error: the requested field " << name << " has a different size than"
388  << " the provided data field.\n";
389  errorstr << "Registerd as: " << info.iSize() << "x" << info.jSize() << "x" << info.kSize()
390  << "x" << info.lSize() << "\n";
391  errorstr << "Given : " << size.iSize() << "x" << size.jSize() << "x" << size.kSize()
392  << "x" << 1 << "\n";
393  SerializationException exception;
394  exception.Init(errorstr.str());
395  throw exception;
396  }
397 
398  // Boundaries are usually set to 3 even if there is storage in that direction
399  // Correct this interesting yet totally nonsense behavior
400  const int actual_iminus = hasStorageInI ? boundary.iMinusOffset() : 0;
401  const int actual_jminus = hasStorageInJ ? boundary.jMinusOffset() : 0;
402  const int actual_iplus = hasStorageInI ? boundary.iPlusOffset() : 0;
403  const int actual_jplus = hasStorageInJ ? boundary.jPlusOffset() : 0;
404 
405  if((info.iMinusHaloSize() != -actual_iminus) || (info.jMinusHaloSize() != -actual_jminus) ||
406  (info.kMinusHaloSize() != -boundary.kMinusOffset()) || (info.lMinusHaloSize() != -0) ||
407  (info.iPlusHaloSize() != actual_iplus) || (info.jPlusHaloSize() != actual_jplus) ||
408  (info.kPlusHaloSize() != boundary.kPlusOffset()) || (info.lPlusHaloSize() != 0)) {
409  // Throw exception
410  std::ostringstream errorstr;
411  errorstr << "Error: the requested field " << name << " has different boundaries than"
412  << " the provided data field.";
413  SerializationException exception;
414  exception.Init(errorstr.str());
415  throw exception;
416  }
417 
418  // Check data type
419  if(info.type() != type_name<ValueType>()) {
420  // Throw exception
421  std::ostringstream errorstr;
422  errorstr << "Error: the requested field " << name << " has different type than"
423  << " the provided data field (expected " << info.type() << ", got "
424  << type_name<ValueType>() << ")";
425  SerializationException exception;
426  exception.Init(errorstr.str());
427  throw exception;
428  }
429 
430  // This hack is used to retrieve the storage pointer for the host memory in case of GPU fields
431  // This will synchronize the host storage
432  ValueType* data =
433  &field(boundary.iMinusOffset(), boundary.jMinusOffset(), boundary.kMinusOffset());
434 
435  // Perform the read
436  this->ReadField(name, savepoint, data, iStride, jStride, kStride, 0, alsoPrevious);
437 }

441 template <typename TDataField>
442 void Serializer::InitializeField(const std::string& fieldname, TDataField& field,
443  bool hasStorageInI, bool hasStorageInJ) const {
444  // Get info (will throw if does not exist)
445  const serialbox::stella::DataFieldInfo& info = FindField(fieldname);
446 
447  const int ct_iminus = TDataField::StorageFormat::IJBoundary::IMinusOffset::value;
448  const int ct_iplus = TDataField::StorageFormat::IJBoundary::IPlusOffset::value;
449  const int ct_jminus = TDataField::StorageFormat::IJBoundary::JMinusOffset::value;
450  const int ct_jplus = TDataField::StorageFormat::IJBoundary::JPlusOffset::value;
451 
452  const bool ok_i = (info.iMinusHaloSize() == (hasStorageInI ? -ct_iminus : 0)) &&
453  (info.iPlusHaloSize() == (hasStorageInI ? ct_iplus : 0));
454  const bool ok_j = (info.jMinusHaloSize() == (hasStorageInJ ? -ct_jminus : 0)) &&
455  (info.jPlusHaloSize() == (hasStorageInJ ? ct_jplus : 0));
456 
457  // Check IJ boundary
458  if(!(ok_i && ok_j)) {
459  std::stringstream errorstr;
460  errorstr << "Error: the requested field " << fieldname << " has a different ij-boundary than"
461  << " what the provided data field object supports.\n";
462  serialbox::stella::SerializationException exception;
463  exception.Init(errorstr.str());
464  throw exception;
465  }
466 
467  // Put together size and k boundary (Note that IJKSize and KBoundary come from STELLA!)
468  ::IJKSize calculationDomain;
469  calculationDomain.Init(info.calculationDomain().iSize(), info.calculationDomain().jSize(),
470  info.calculationDomain().kSize());
471  ::KBoundary kBoundary;
472  kBoundary.Init(info.kMinusHaloSize(), info.kPlusHaloSize());
473 
474  // Initialize field
475  field.Init(fieldname, calculationDomain, kBoundary);
476 }

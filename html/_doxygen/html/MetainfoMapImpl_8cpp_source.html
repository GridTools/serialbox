25 struct InsertHelper {
26  // Capture environment
27  MetainfoMapImpl& map;
28  const std::string& key;

31  // Read value from JSON node (and check if the types match) and insert it into the MetainfoMapImpl
32  // as
33  // type ´T´
34  template <class T, class CheckFunction>
35  void insertAs(CheckFunction&& checkFunction, const char* valueStr) {
36  if(!(node["value"].*checkFunction)())
37  throw Exception("sub-node '%s' not regconized as %s", key, valueStr);
38  T value = node["value"];
39  map.insert(key, value);
40  }
41 
42  // Read value from JSON node as Array of type ´T´ and insert as array of type ´T´ into the
43  // MetainfoMapImpl
44  template <class T>
45  void insertAsArrayOf() {
46  Array<T> array = node["value"];
47  map.insert(key, array);
48  }

86 json::json MetainfoMapImpl::toJSON() const {
87  json::json jsonNode;
88 
89  if(map_.empty())
90  return jsonNode;
91 
92  for(auto it = map_.cbegin(), end = map_.cend(); it != end; ++it) {
93  const MetainfoValueImpl& value = it->second;
94  const std::string& key = it->first;
95 
96  jsonNode[key]["type_id"] = static_cast<int>(value.type());
97 
98  json::json valueNode;
99  const bool isArray = TypeUtil::isArray(value.type());
100 
101  switch(TypeUtil::getPrimitive(value.type())) {
102  case TypeID::Boolean:
103  if(isArray) {
104  for(const bool& v : value.as<Array<bool>>())
105  valueNode.push_back(v);
106  } else {
107  valueNode = value.as<bool>();
108  }
109  break;
110  case TypeID::Int32:
111  if(isArray) {
112  for(const int& v : value.as<Array<int>>())
113  valueNode.push_back(v);
114  } else {
115  valueNode = value.as<int>();
116  }
117  break;
118  case TypeID::Int64:
119  if(isArray) {
120  for(const std::int64_t& v : value.as<Array<std::int64_t>>())
121  valueNode.push_back(v);
122  } else {
123  valueNode = value.as<std::int64_t>();
124  }
125  break;
126  case TypeID::Float32:
127  if(isArray) {
128  for(const float& v : value.as<Array<float>>())
129  valueNode.push_back(v);
130  } else {
131  valueNode = value.as<float>();
132  }

160 void MetainfoMapImpl::fromJSON(const json::json& jsonNode) {
161  map_.clear();
162 
163  if(jsonNode.is_null() || jsonNode.empty())
164  return;
165 
166  for(auto it = jsonNode.begin(), end = jsonNode.end(); it != end; ++it) {
167 
168  if(!it->count("type_id"))
169  throw Exception("sub-node '%s' has no node 'type_id'", it.key());
170 
171  if(!it->count("value"))
172  throw Exception("sub-node '%s' has no node 'value'", it.key());
173 
174  const json::json& node = it.value();
175  const std::string& key = it.key();
176  const int typeAsInt = node["type_id"];
177  const TypeID type = static_cast<TypeID>(typeAsInt);
178  const bool isArray = TypeUtil::isArray(type);
179 
180  InsertHelper insertHelper{*this, key, node};
181 
182  switch(TypeUtil::getPrimitive(type)) {
183  case TypeID::Boolean:
184  if(isArray) {
185  insertHelper.insertAsArrayOf<bool>();
186  } else {
187  insertHelper.insertAs<bool>(&json::json::is_boolean, "boolean");
188  }
189  break;
190  case TypeID::Int32:
191  if(isArray) {
192  insertHelper.insertAsArrayOf<int>();
193  } else {
194  insertHelper.insertAs<int>(&json::json::is_number_integer, "integer");
195  }
196  break;
197  case TypeID::Int64:
198  if(isArray) {
199  insertHelper.insertAsArrayOf<std::int64_t>();
200  } else {
201  insertHelper.insertAs<std::int64_t>(&json::json::is_number_integer, "integer");
202  }
203  break;
204  case TypeID::Float32:
205  if(isArray) {
206  insertHelper.insertAsArrayOf<float>();
207  } else {
208  insertHelper.insertAs<float>(&json::json::is_number, "floating pointer number");
209  }
210  break;
211  case TypeID::Float64:
212  if(isArray) {
213  insertHelper.insertAsArrayOf<double>();
214  } else {
215  insertHelper.insertAs<double>(&json::json::is_number, "floating pointer number");
216  }
217  break;
218  case TypeID::String:
219  if(isArray) {
220  insertHelper.insertAsArrayOf<std::string>();
221  } else {
222  insertHelper.insertAs<std::string>(&json::json::is_string, "string");
223  }
224  break;
225  default:
226  serialbox_unreachable("Invalid TypeID");
227  }
228  }
229 }

30 class BinaryBuffer {
31 public:
33  BinaryBuffer(const StorageView& storageView) {
34  const auto& slice = storageView.getSlice();
35 
36  if(slice.empty()) {
37  buffer_.resize(storageView.sizeInBytes());
38  offset_ = 0;
39  } else {
40  const auto& dims = storageView.dims();
41  const auto& triple = slice.sliceTriples().back();
42  const int bytesPerElement = storageView.bytesPerElement();
43 
44  // Allocate a buffer which can be efficently loaded. The buffer will treat the
45  // dimensions dim_{1}, ..., dim_{N-1} as full while last the dimension dim_{N} as sliced but
46  // without incorporating the step. This is necessary as we only want to call ::write once.
47 
48  // Compute dimensions
49  dims_ = dims;
50  dims_.back() = triple.stop - triple.start;
51 
52  // Compute strides (col-major)
53  strides_.resize(dims_.size());
54 
55  int stride = 1;
56  strides_[0] = stride;
57 
58  for(int i = 1; i < dims_.size(); ++i) {
59  stride *= dims_[i - 1];
60  strides_[i] = stride;
61  }
62 
63  // Compute size
64  std::size_t size = 1;
65  for(std::size_t i = 0; i < dims_.size(); ++i)
66  size *= (dims_[i] == 0 ? 1 : dims_[i]);
67 
68  // Compute initial offset in bytes
69  offset_ = (strides_.back() * triple.start) * bytesPerElement;
70 
71  buffer_.resize(size * bytesPerElement);
72  }
73  }

76  void copyBufferToStorageView(StorageView& storageView) {
77  const auto& slice = storageView.getSlice();
78 
79  if(slice.empty()) {
80  Byte* dataPtr = buffer_.data();
81  const int bytesPerElement = storageView.bytesPerElement();
82 
83  if(storageView.isMemCopyable()) {
84  std::memcpy(storageView.originPtr(), dataPtr, buffer_.size());
85  } else {
86  for(auto it = storageView.begin(), end = storageView.end(); it != end;
87  ++it, dataPtr += bytesPerElement)
88  std::memcpy(it.ptr(), dataPtr, bytesPerElement);
89  }
90 
91  } else {
92  const int numDims = dims_.size();
93  const auto& triples = slice.sliceTriples();
94  const int bytesPerElement = storageView.bytesPerElement();
95  Byte* dataPtr = buffer_.data();
96 
97  // Compute intial indices in the buffer
98  std::vector<int> index(numDims);
99  for(int i = 0; i < numDims - 1; ++i)
100  index[i] = triples[i].start;
101  index.back() = 0;
102 
103  // Iterate over the the storageView and the Buffer
104  Byte* curPtr = buffer_.data();
105  for(auto it = storageView.begin(), end = storageView.end(); it != end; ++it) {
106 
107  // Compute position of current element
108  int pos = 0;
109  for(int i = 0; i < numDims; ++i)
110  pos += bytesPerElement * (strides_[i] * index[i]);
111  curPtr = dataPtr + pos;
112 
113  // Memcopy the current elemment to the storageView
114  std::memcpy(it.ptr(), curPtr, bytesPerElement);
115 
116  // Compute the index of the next element in the buffer
117  for(int i = 0; i < numDims; ++i)
118  if((index[i] += triples[i].step) < triples[i].stop)
119  break;
120  else
121  index[i] = triples[i].start;
122  }
123  }
124  }

128  Byte* dataPtr = buffer_.data();
129  const int bytesPerElement = storageView.bytesPerElement();
130 
131  if(storageView.isMemCopyable()) {
132  std::memcpy(dataPtr, storageView.originPtr(), buffer_.size());
133  } else {
134  for(auto it = storageView.begin(), end = storageView.end(); it != end;
135  ++it, dataPtr += bytesPerElement)
136  std::memcpy(dataPtr, it.ptr(), bytesPerElement);
137  }
138  }

166 BinaryArchive::BinaryArchive(OpenModeKind mode, const std::string& directory,
167  const std::string& prefix, bool skipMetaData)
168  : mode_(mode), directory_(directory), prefix_(prefix), json_() {
169 
170  LOG(info) << "Creating BinaryArchive (mode = " << mode_ << ") from directory " << directory_;
171 
172  metaDatafile_ = directory_ / ("ArchiveMetaData-" + prefix_ + ".json");
173  hash_ = HashFactory::create(HashFactory::defaultHash());
174 
175  try {
176  bool isDir = boost::filesystem::is_directory(directory_);
177 
178  switch(mode_) {
179  // We are reading, the directory needs to exist
180  case OpenModeKind::Read:
181  if(!isDir)
182  throw Exception("no such directory: '%s'", directory_.string());
183  break;
184  // We are writing or appending, create directories if it they don't exist
185  case OpenModeKind::Write:
186  case OpenModeKind::Append:
187  if(!isDir)
188  boost::filesystem::create_directories(directory_);
189  break;
190  }
191  } catch(boost::filesystem::filesystem_error& e) {
192  throw Exception(e.what());
193  }
194 
195  if(!skipMetaData)
196  readMetaDataFromJson();
197 
198  // Remove all files
199  if(mode_ == OpenModeKind::Write)
200  clear();
201 }

205 void BinaryArchive::readMetaDataFromJson() {
206  LOG(info) << "Reading MetaData for BinaryArchive ... ";
207 
208  // Check if metaData file exists
209  if(!boost::filesystem::exists(metaDatafile_)) {
210  if(mode_ != OpenModeKind::Read)
211  return;
212  throw Exception("archive meta data not found in directory '%s'", directory_.string());
213  }
214 
215  std::ifstream fs(metaDatafile_.string(), std::ios::in);
216  fs >> json_;
217  fs.close();
218 
219  int serialboxVersion = json_["serialbox_version"];
220  std::string archiveName = json_["archive_name"];
221  int archiveVersion = json_["archive_version"];
222  std::string hashAlgorithm = json_["hash_algorithm"];
223 
224  // Check consistency
225  if(!Version::match(serialboxVersion))
226  throw Exception("serialbox version of binary archive (%s) does not match the version "
227  "of the library (%s)",
228  Version::toString(serialboxVersion), SERIALBOX_VERSION_STRING);
229 
230  if(archiveName != BinaryArchive::Name)
231  throw Exception("archive is not a binary archive");
232 
233  if(archiveVersion != BinaryArchive::Version)
234  throw Exception("binary archive version (%s) does not match the version of the library (%s)",
235  archiveVersion, BinaryArchive::Version);
236 
237  // Set the correct hash algorithm
238  hash_ = HashFactory::create(hashAlgorithm);
239 
240  // Deserialize FieldTable
241  for(auto it = json_["fields_table"].begin(); it != json_["fields_table"].end(); ++it) {
242  FieldOffsetTable fieldOffsetTable;
243 
244  // Iterate over savepoint of this field
245  for(auto fileOffsetIt = it->begin(); fileOffsetIt != it->end(); ++fileOffsetIt)
246  fieldOffsetTable.push_back(FileOffsetType{fileOffsetIt->at(0), fileOffsetIt->at(1)});
247 
248  fieldTable_[it.key()] = fieldOffsetTable;
249  }
250 }

287 FieldID BinaryArchive::write(const StorageView& storageView, const std::string& field,
288  const std::shared_ptr<FieldMetainfoImpl> info) throw(Exception) {
289  if(mode_ == OpenModeKind::Read)
290  throw Exception("Archive is not initialized with OpenModeKind set to 'Write' or 'Append'");
291 
292  LOG(info) << "Attempting to write field \"" << field << "\" to BinaryArchive ...";
293 
294  boost::filesystem::path filename(directory_ / (prefix_ + "_" + field + ".dat"));
295  std::ofstream fs;
296 
297  // Create binary data buffer
298  BinaryBuffer binaryBuffer(storageView);
299  binaryBuffer.copyStorageViewToBuffer(storageView);
300 
301  // Compute hash
302  std::string checksum(hash_->hash(binaryBuffer.data(), binaryBuffer.size()));
303 
304  // Check if field already exists
305  auto it = fieldTable_.find(field);
306  FieldID fieldID{field, 0};
307 
308  // Field does exists
309  if(it != fieldTable_.end()) {
310  FieldOffsetTable& fieldOffsetTable = it->second;
311 
312  // Check if field has already been serialized by comparing the checksum
313  for(std::size_t i = 0; i < fieldOffsetTable.size(); ++i)
314  if(checksum == fieldOffsetTable[i].checksum) {
315  LOG(info) << "Field \"" << field << "\" already serialized (id = " << i << "). Stopping";
316  fieldID.id = i;
317  return fieldID;
318  }
319 
320  // Append field at the end
321  fs.open(filename.string(), std::ofstream::out | std::ofstream::binary | std::ofstream::app);
322 #ifdef SERIALBOX_COMPILER_MSVC
323  fs.seekp(0, fs.end);
324 #endif
325  auto offset = fs.tellp();
326  fieldID.id = fieldOffsetTable.size();
327  fieldOffsetTable.push_back(FileOffsetType{offset, checksum});
328 
329  LOG(info) << "Appending field \"" << fieldID.name << "\" (id = " << fieldID.id << ") to "
330  << filename.filename();
331  }
332  // Field does not exist, create new file and append data
333  else {
334  fs.open(filename.string(), std::ios::out | std::ios::binary | std::ios::trunc);
335  fieldID.id = 0;
336 
337  fieldTable_.insert(
338  FieldTable::value_type(fieldID.name, FieldOffsetTable(1, FileOffsetType{0, checksum})));
339 
340  LOG(info) << "Creating new file " << filename.filename() << " for field \"" << fieldID.name
341  << "\" (id = " << fieldID.id << ")";
342  }
343 
344  if(!fs.is_open())
345  throw Exception("cannot open file: '%s'", filename.string());
346 
347  // Write binaryData to disk
348  fs.write(binaryBuffer.data(), binaryBuffer.size());
349  fs.close();
350 
351  updateMetaData();
352 
353  LOG(info) << "Successfully wrote field \"" << fieldID.name << "\" (id = " << fieldID.id << ") to "
354  << filename.filename();
355  return fieldID;
356 }

377 void BinaryArchive::read(StorageView& storageView, const FieldID& fieldID,
378  std::shared_ptr<FieldMetainfoImpl> info) const throw(Exception) {
379  if(mode_ != OpenModeKind::Read)
380  throw Exception("Archive is not initialized with OpenModeKind set to 'Read'");
381 
382  LOG(info) << "Attempting to read field \"" << fieldID.name << "\" (id = " << fieldID.id
383  << ") via BinaryArchive ... ";
384 
385  // Check if field exists
386  auto it = fieldTable_.find(fieldID.name);
387  if(it == fieldTable_.end())
388  throw Exception("no field '%s' registered in BinaryArchive", fieldID.name);
389 
390  const FieldOffsetTable& fieldOffsetTable = it->second;
391 
392  // Check if id is valid
393  if(fieldID.id >= fieldOffsetTable.size())
394  throw Exception("invalid id '%i' of field '%s'", fieldID.id, fieldID.name);
395 
396  // Create binary data buffer
397  BinaryBuffer binaryBuffer(storageView);
398 
399  // Open file & read into binary buffer
400  std::string filename((directory_ / (prefix_ + "_" + fieldID.name + ".dat")).string());
401  std::ifstream fs(filename, std::ios::binary);
402 
403  if(!fs.is_open())
404  throw Exception("cannot open file: '%s'", filename);
405 
406  // Set position in the stream
407  auto offset = fieldOffsetTable[fieldID.id].offset + binaryBuffer.offset();
408  fs.seekg(offset);
409 
410  // Read data into contiguous memory
411  fs.read(binaryBuffer.data(), binaryBuffer.size());
412  fs.close();
413 
414  binaryBuffer.copyBufferToStorageView(storageView);
415 
416  LOG(info) << "Successfully read field \"" << fieldID.name << "\" (id = " << fieldID.id << ")";
417 }

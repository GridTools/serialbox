34 SerializerImpl::SerializerImpl(OpenModeKind mode, const std::string& directory,
35  const std::string& prefix, const std::string& archiveName)

40  enabled_ = (envvar && std::atoi(envvar) > 0) ? -1 : 1;
41  }

57  }

61  if(!upgradeMetaData()) {
62  constructMetaDataFromJson();
63  constructArchive(archiveName);
64  }
65 
66  // If mode is writing drop all files

69 }

86 static inline bool dimsEqual(const std::vector<int>& dims1, const std::vector<int>& dims2) {
87  if(dims1.size() != dims2.size())
88  return false;
89 
90  // If a dimensions is negative, 0 or 1 it is ignored. We have to do this as gridtools treats empty
91  // dimensions as 0 while STELLA and Frotran usually set them to 1.
92  for(std::size_t i = 0; i < dims1.size(); ++i)
93  if(dims1[i] != dims2[i] && !(dims1[i] <= 1 && dims2[i] <= 1))
94  return false;
95  return true;
96 }

98 std::shared_ptr<FieldMetainfoImpl>
99 SerializerImpl::checkStorageView(const std::string& name, const StorageView& storageView) const {
100 
101  // Check if field exists
102  auto fieldIt = fieldMap_->findField(name);
103  if(fieldIt == fieldMap_->end())
104  throw Exception("field '%s' is not registerd within the Serializer", name);
105 
106  const FieldMetainfoImpl& fieldInfo = *fieldIt->second;
107 
108  // Check if types match
109  if(fieldInfo.type() != storageView.type())
110  throw Exception("field '%s' has type '%s' but was registrered as type '%s'", name,
111  TypeUtil::toString(fieldInfo.type()), TypeUtil::toString(storageView.type()));
112 
113  // Check if dimensions match
114  if(!dimsEqual(fieldInfo.dims(), storageView.dims())) {
115  throw Exception("dimensions of field '%s' do not match regsitered ones:"
116  "\nRegistred as: [ %s ]"
117  "\nGiven as: [ %s ]",
118  name, ArrayUtil::toString(fieldInfo.dims()),
119  ArrayUtil::toString(storageView.dims()));
120  }
121  return fieldIt->second;
122 }

128 void SerializerImpl::write(const std::string& name, const SavepointImpl& savepoint,
129  const StorageView& storageView) {
130  if(SerializerImpl::serializationStatus() < 0)
131  return;
132 
133  LOG(info) << "Serializing field \"" << name << "\" at savepoint \"" << savepoint << "\" ... ";
134 
135  if(mode_ == OpenModeKind::Read)
136  throw Exception("serializer not open in write mode, but write operation requested");
137 
138  //
139  // 1) Check if field is registred within the Serializer and perform some consistency checks
140  //
141  auto info = checkStorageView(name, storageView);
142 
143  //
144  // 2) Locate savepoint and register it if necessary
145  //
146  int savepointIdx = savepointVector_->find(savepoint);
147 
148  if(savepointIdx == -1) {
149  LOG(info) << "Registering new savepoint \"" << savepoint << "\"";
150  savepointIdx = savepointVector_->insert(savepoint);
151  }
152 
153  //
154  // 3) Check if field can be added to Savepoint
155  //
156  if(savepointVector_->hasField(savepointIdx, name))
157  throw Exception("field '%s' already saved at savepoint '%s'", name,
158  (*savepointVector_)[savepointIdx].toString());
159 
160  //
161  // 4) Pass the StorageView to the backend Archive and perform actual data-serialization.
162  //
163  FieldID fieldID = archive_->write(storageView, name, info);
164 
165  //
166  // 5) Register FieldID within Savepoint.
167  //
168  savepointVector_->addField(savepointIdx, fieldID);
169 
170  //
171  // 6) Update meta-data on disk
172  //
173  updateMetaData();
174 
175  LOG(info) << "Successfully serialized field \"" << name << "\"";
176 }

182 void SerializerImpl::read(const std::string& name, const SavepointImpl& savepoint,

185  return;

193  // 1) Check if field is registred within the Serializer and perform some consistency checks

195  auto info = checkStorageView(name, storageView);
196 
197  //
198  // 2) Check if savepoint exists and obtain fieldID
199  //
200  int savepointIdx = savepointVector_->find(savepoint);
201

207  //

211 
212  LOG(info) << "Successfully deserialized field \"" << name << "\"";

240  StorageView& storageView) {
241 #ifdef SERIALBOX_ASYNC_API

271 void SerializerImpl::constructMetaDataFromJson() {
272  LOG(info) << "Constructing Serializer from MetaData ... ";
273 
274  // Try open meta-data file
275  if(!boost::filesystem::exists(metaDataFile_)) {
276  if(mode_ != OpenModeKind::Read)
277  return;
278  else
279  throw Exception("cannot create Serializer: MetaData-%s.json not found in %s", prefix_,
280  directory_);
281  }
282 
283  json::json jsonNode;
284  try {
285  std::ifstream fs(metaDataFile_.string(), std::ios::in);
286  fs >> jsonNode;
287  fs.close();
288  } catch(std::exception& e) {
289  throw Exception("JSON parser error: %s", e.what());
290  }
291 
292  try {
293  // Check consistency
294  if(!jsonNode.count("serialbox_version"))
295  throw Exception("node 'serialbox_version' not found");
296 
297  int serialboxVersion = jsonNode["serialbox_version"];
298 
299  if(!Version::match(serialboxVersion))
300  throw Exception(
301  "serialbox version of MetaData (%s) does not match the version of the library (%s)",
302  Version::toString(serialboxVersion), SERIALBOX_VERSION_STRING);
303 
304  // Check if prefix match
305  if(!jsonNode.count("prefix"))
306  throw Exception("node 'prefix' not found");
307 
308  if(jsonNode["prefix"] != prefix_)
309  throw Exception("inconsistent prefixes: expected '%s' got '%s'", jsonNode["prefix"], prefix_);
310 
311  // Construct globalMetainfo
312  if(jsonNode.count("global_meta_info"))
313  globalMetainfo_->fromJSON(jsonNode["global_meta_info"]);
314 
315  // Construct Savepoints
316  if(jsonNode.count("savepoint_vector"))
317  savepointVector_->fromJSON(jsonNode["savepoint_vector"]);
318 
319  // Construct FieldMap
320  if(jsonNode.count("field_map"))
321  fieldMap_->fromJSON(jsonNode["field_map"]);
322 
323  } catch(Exception& e) {
324  throw Exception("error while parsing %s: %s", metaDataFile_, e.what());
325  }
326 }

399 bool SerializerImpl::upgradeMetaData() {
400  boost::filesystem::path oldMetaDataFile = directory_ / (prefix_ + ".json");
401 
402  //
403  // Check if upgrade is necessary
404  //
405 
406  try {
407  // Check if prefix.json exists
408  if(!boost::filesystem::exists(oldMetaDataFile))
409  return false;
410 
411  LOG(info) << "Detected old serialbox meta-data " << oldMetaDataFile;
412 
413  // Check if we already upgraded this archive
414  if(boost::filesystem::exists(metaDataFile_) &&
415  (boost::filesystem::last_write_time(oldMetaDataFile) <
416  boost::filesystem::last_write_time(metaDataFile_))) {
417  return false;
418  }
419  } catch(boost::filesystem::filesystem_error& e) {
420  throw Exception("filesystem error: %s", e.what());
421  }
422 
423  LOG(info) << "Upgrading meta-data to serialbox version (" << SERIALBOX_VERSION_STRING << ") ...";
424 
425  if(mode_ != OpenModeKind::Read)
426  throw Exception("old serialbox archives cannot be opened in 'Write' or 'Append' mode");
427 
428  json::json oldJson;
429  std::ifstream ifs(oldMetaDataFile.string());
430  if(!ifs.is_open())
431  throw Exception("upgrade failed: cannot open %s", oldMetaDataFile);
432  ifs >> oldJson;
433  ifs.close();
434 
435  //
436  // Upgrade Metainfo
437  //
438 
439  // Try to guess the precision of the floating point type. We try to match the floating point type
440  // of the fields while defaulting to double.
441  TypeID globalMetainfoFloatType = TypeID::Float64;
442  if(oldJson.count("FieldsTable") && oldJson["FieldsTable"].size() > 0) {
443  if(oldJson["FieldsTable"][0]["__elementtype"] == "float")
444  globalMetainfoFloatType = TypeID::Float32;
445  }
446 
447  LOG(info) << "Deduced float type of global meta-info as: " << globalMetainfoFloatType;
448 
449  if(oldJson.count("GlobalMetainfo")) {
450 
451  LOG(info) << "Upgrading global meta-info ...";
452 
453  for(auto it = oldJson["GlobalMetainfo"].begin(), end = oldJson["GlobalMetainfo"].end();
454  it != end; ++it) {
455 
456  LOG(info) << "Inserting global meta-info: key = " << it.key() << ", value = " << it.value();
457 
458  std::string key = it.key();
459  if(!boost::algorithm::starts_with(key, "__")) {
460  if(it.value().is_string()) {
461  std::string value = it.value();
462  addGlobalMetainfo(key, value);
463  } else if(it.value().is_boolean()) {
464  addGlobalMetainfo(key, bool(it.value()));
465  } else if(it.value().is_number_integer()) {
466  addGlobalMetainfo(key, int(it.value()));
467  } else if(it.value().is_number_float()) {
468  if(globalMetainfoFloatType == TypeID::Float32)
469  addGlobalMetainfo(key, float(it.value()));
470  else
471  addGlobalMetainfo(key, double(it.value()));
472  } else
473  throw Exception("failed to upgrade: cannot deduce type of globalMetainfo '%s'", it.key());
474  }
475  }
476 
477  LOG(info) << "Successfully upgraded global meta-info";
478  }
479 
480  //
481  // Upgrade FieldsTable
482  //
483 
484  if(oldJson.count("FieldsTable")) {
485 
486  LOG(info) << "Upgrading fields table ...";
487 
488  const auto& fieldsTable = oldJson["FieldsTable"];
489  for(std::size_t i = 0; i < fieldsTable.size(); ++i) {
490  auto& fieldInfo = fieldsTable[i];
491  std::string name = fieldInfo["__name"];
492 
493  LOG(info) << "Inserting field: " << name;
494 
495  // Get Type
496  std::string elementtype = fieldInfo["__elementtype"];
497  TypeID type = TypeID::Float64;
498 
499  if(elementtype == "int")
500  type = TypeID::Int32;
501  else if(elementtype == "float")
502  type = TypeID::Float32;
503  else if(elementtype == "double")
504  type = TypeID::Float64;
505 
506  // Get dimension
507  std::vector<int> dims(3, 1);
508  dims[0] = int(fieldInfo["__isize"]);
509  dims[1] = int(fieldInfo["__jsize"]);
510  dims[2] = int(fieldInfo["__ksize"]);
511 
512  if(fieldInfo.count("__lsize"))
513  dims.push_back(int(fieldInfo["__lsize"]));
514 
515  // Iterate field meta-info
516  MetainfoMapImpl metaInfo;
517  for(auto it = fieldInfo.begin(), end = fieldInfo.end(); it != end; ++it) {
518  std::string key = it.key();
519  if(it.value().is_string()) {
520  std::string value = it.value();
521  metaInfo.insert(key, value);
522  } else if(it.value().is_boolean()) {
523  metaInfo.insert(key, bool(it.value()));
524  } else if(it.value().is_number_integer()) {
525  metaInfo.insert(key, int(it.value()));
526  } else if(it.value().is_number_float()) {
527  if(globalMetainfoFloatType == TypeID::Float32)
528  metaInfo.insert(key, float(it.value()));
529  else
530  metaInfo.insert(key, double(it.value()));
531  } else
532  throw Exception("failed to upgrade: Cannot deduce type of meta-info '%s' of field '%s'",
533  key, name);
534  }
535 
536  fieldMap_->insert(name, type, dims, metaInfo);
537  }
538 
539  LOG(info) << "Successfully upgraded fields table";
540  }
541 
542  //
543  // Upgrade SavepointVector and ArchiveMetaData
544  //
545 
546  // Construct archive but don't parse the meta-data (we will do it ourselves below)
547  archive_ = std::make_unique<BinaryArchive>(mode_, directory_.string(), prefix_, true);
548 
549  // Old serialbox always uses SHA256
550  static_cast<BinaryArchive*>(archive_.get())->setHash(HashFactory::create("SHA256"));
551 
552  BinaryArchive::FieldTable& fieldTable = static_cast<BinaryArchive*>(archive_.get())->fieldTable();
553 
554  if(oldJson.count("OffsetTable")) {
555 
556  LOG(info) << "Upgrading offset table ...";
557 
558  const auto& offsetTable = oldJson["OffsetTable"];
559  for(std::size_t i = 0; i < offsetTable.size(); ++i) {
560  auto& offsetTableEntry = offsetTable[i];
561 
562  // Create savepoint
563  std::string name = offsetTableEntry["__name"];
564  SavepointImpl savepoint(name);
565 
566  // Add meta-info to savepoint
567  for(auto it = offsetTableEntry.begin(), end = offsetTableEntry.end(); it != end; ++it) {
568  std::string key = it.key();
569  if(!boost::algorithm::starts_with(key, "__")) {
570  if(it.value().is_string()) {
571  std::string value = it.value();
572  savepoint.addMetainfo(it.key(), value);
573  } else if(it.value().is_boolean()) {
574  savepoint.addMetainfo(it.key(), bool(it.value()));
575  } else if(it.value().is_number_integer()) {
576  savepoint.addMetainfo(it.key(), int(it.value()));
577  } else if(it.value().is_number_float()) {
578  if(globalMetainfoFloatType == TypeID::Float32)
579  savepoint.addMetainfo(it.key(), float(it.value()));
580  else
581  savepoint.addMetainfo(it.key(), double(it.value()));
582  } else
583  throw Exception(
584  "failed to upgrade: Cannot deduce type of meta-info '%s' of savepoint '%s'",
585  it.key(), name);
586  }
587  }
588 
589  LOG(info) << "Adding savepoint: " << savepoint;
590 
591  // Register savepoint
592  int savepointIdx = savepointVector_->insert(savepoint);
593  assert(savepointIdx != -1);
594 
595  // Add fields to savepoint and field table of the archive
596  for(auto it = offsetTableEntry["__offsets"].begin(),
597  end = offsetTableEntry["__offsets"].end();
598  it != end; ++it) {
599  std::string fieldname = it.key();
600 
601  FieldID fieldID{fieldname, 0};
602  BinaryArchive::FileOffsetType fileOffset{it.value()[0], it.value()[1]};
603 
604  // Insert offsets into the field table (This mimics the write operation of the
605  // Binary archive)
606  auto fieldTableIt = fieldTable.find(fieldname);
607  if(fieldTableIt != fieldTable.end()) {
608  BinaryArchive::FieldOffsetTable& fieldOffsetTable = fieldTableIt->second;
609  bool fieldAlreadySerialized = false;
610 
611  // Check if field has already been serialized by comparing the checksum
612  for(std::size_t i = 0; i < fieldOffsetTable.size(); ++i)
613  if(fileOffset.checksum == fieldOffsetTable[i].checksum) {
614  fieldAlreadySerialized = true;
615  fieldID.id = i;
616  break;
617  }
618 
619  // Append field at the end
620  if(!fieldAlreadySerialized) {
621  assert(fileOffset.offset != 0);
622  fieldID.id = fieldOffsetTable.size();
623  fieldOffsetTable.push_back(fileOffset);
624  }
625  } else {
626  assert(fileOffset.offset == 0);
627  fieldID.id = 0;
628  fieldTable.insert(BinaryArchive::FieldTable::value_type(
629  fieldname, BinaryArchive::FieldOffsetTable(1, fileOffset)));
630  }
631 
632  // Add field to savepoint
633  savepointVector_->addField(savepointIdx, fieldID);
634 
635  LOG(info) << "Adding field '" << fieldID << "' to savepoint " << savepoint;
636  }
637  }
638 
639  LOG(info) << "Successfully upgraded offset table";
640  }
641 
642  // Try to write the mata data to disk so that we can avoid such an upgrade in the future. However,
643  // if we read from a location where we have no write permission, this should be non-fatal.
644  try {
645  updateMetaData();
646  } catch(Exception& e) {
647  LOG(warning) << "Failed to write upgraded meta-data to disk: " << e.what();
648  }
649 
650  LOG(info) << "Successfully upgraded MetaData to serialbox version (" << SERIALBOX_VERSION_STRING
651  << ")";
652 
653  return true;
654 }

55 template <class ArrayType1, class ArrayType2>
56 static void checkRank(const char* name, ArrayType1&& array, ArrayType2&& arrayRef) {
57  const int rank = (array[0] > 0 ? 1 : 0) + (array[1] > 0 ? 1 : 0) + (array[2] > 0 ? 1 : 0) +
58  (array[3] > 0 ? 1 : 0);
59 
60  const int refRank = (arrayRef[0] > 0 ? 1 : 0) + (arrayRef[1] > 0 ? 1 : 0) +
61  (arrayRef[2] > 0 ? 1 : 0) + (arrayRef[3] > 0 ? 1 : 0);
62 
63  if(rank != refRank)
64  throw Exception("field '%s' has rank %i but field with rank %i was passed", name, refRank,
65  rank);
66 }

68 void serialboxFortranSerializerCheckField(const void* serializer, const char* name, int* type,
69  int* isize, int* jsize, int* ksize, int* lsize) {
70 
71  const Serializer* ser = toConstSerializer(static_cast<const serialboxSerializer_t*>(serializer));
72 
73  try {
74  const auto& info = ser->getFieldMetainfoImplOf(name);
75 
76  if(info.dims().size() != 4)
77  throw Exception("number of dimensions is %i, required are 4");
78 
79  std::array<int, 4> actualSizes{{*isize, *jsize, *ksize, *lsize}};
80  const auto& refSizes = info.dims();
81 
82  // Check rank
83  checkRank(name, actualSizes, refSizes);
84 
85  // Check type (be careful with converting *type as it is an arbitrary int)
86  TypeID typeID = *type <= Float64 ? (TypeID)*type : TypeID::Invalid;
87  if(typeID != info.type())
88  throw Exception("field '%s' has type '%s' but was registrered as type '%s'", name,
89  serialbox::TypeUtil::toString(info.type()),
90  serialbox::TypeUtil::toString(typeID));
91 
92  // Reorder and check dimensions
93  for(int i = 0; i < 4; ++i) {
94  if(actualSizes[i] == refSizes[i])
95  continue;
96 
97  if(refSizes[i] == 1) {
98  for(int j = 3; j > i; --j)
99  actualSizes[j] = actualSizes[j - 1];
100  actualSizes[i] = 1;
101  continue;
102  } else
103  throw Exception("dimensions of field '%s' do not match regsitered ones:"
104  "\nRegistred as: [ %i, %i, %i, %i ]"
105  "\nGiven as: [ %i, %i, %i, %i ]",
106  name, refSizes[0], refSizes[1], refSizes[2], refSizes[3], actualSizes[0],
107  actualSizes[1], actualSizes[2], actualSizes[3]);
108  }
109  } catch(std::exception& e) {
110  serialboxFatalError(e.what());
111  }
112 }

114 void serialboxFortranComputeStrides(void* serializer, const char* fieldname, const void* basePtr,
115  const void* iplus1, const void* jplus1, const void* kplus1,
116  const void* lplus1, int* istride, int* jstride, int* kstride,
117  int* lstride) {
118  Serializer* ser = toSerializer(static_cast<serialboxSerializer_t*>(serializer));
119 
120  try {
121  const auto& info = ser->getFieldMetainfoImplOf(fieldname);
122 
123  std::array<long, 4> strides{
124  {reinterpret_cast<const char*>(iplus1) - reinterpret_cast<const char*>(basePtr),
125  reinterpret_cast<const char*>(jplus1) - reinterpret_cast<const char*>(basePtr),
126  reinterpret_cast<const char*>(kplus1) - reinterpret_cast<const char*>(basePtr),
127  reinterpret_cast<const char*>(lplus1) - reinterpret_cast<const char*>(basePtr)}};
128 
129  if(info.dims().size() != 4)
130  throw Exception("number of dimensions is %i, required are 4");
131 
132  const auto& dims = info.dims();
133 
134  // Check rank
135  checkRank(fieldname, strides, dims);
136 
137  // Reorder strides
138  for(int i = 0; i < 4; ++i) {
139  if(dims[i] <= 1) {
140 
141  // Shift strides to the left and set the current one to 0
142  for(int j = 3; j > i; --j)
143  strides[j] = strides[j - 1];
144 
145  strides[i] = 0;
146  }
147  }
148 
149  // Convert to unit-strides
150  const int bytesPerElement = serialbox::TypeUtil::sizeOf(info.type());
151  *istride = strides[0] / bytesPerElement;
152  *jstride = strides[1] / bytesPerElement;
153  *kstride = strides[2] / bytesPerElement;
154  *lstride = strides[3] / bytesPerElement;
155 
156  } catch(std::exception& e) {
157  serialboxFatalError(e.what());
158  }
159 }
